		(css:php)

!@	BIG_PHP		=	E_PHP|NUMBER|{whitespaces}|{comments}|{operators}|{strings}|{punctuators}
// Attention {comments} avant {operators} : ordre important pour le MultiRegExp Lexer
		whitespaces		=	L_NEW_LINE|TAB|SPACES
		operators		=	PHP_COMPARISON_OP|PHP_ASSIGNMENT_OP|PHP_ARITHMETIC_OP|PHP_LOGICAL_OP|PHP_BITWISE_OP|PHP_ERROR_CONTROL_OP|PHP_STRING_OP|PHP_TYPE_OP
		// Important: pour le MultiRegExp Lexer
		// ASSIGNMENT|ARITHMETIC, COMPARISON|ASSIGNMENT, LOGICAL|BITWISE, ...
		// mais il va planté quand même
		comments		=	S_SLC|S_MLC
		strings			=	S_SSQ|S_SDQ
		punctuators		=	ELISION|LBRACE|RBRACE|LPAREN|RPAREN|LBRACK|RBRACK|DOT|SEMI|QUESTION|COLON

!@	PHP_ID		=	PHP_RESERVED|PHP_LITERAL|PHP_KEYWORD|PHP_SPECIAL_VARS|PHP_FUNCTION|PHP_IDENTIFIER


	E_PHP		=	\?\>
			(css:tag)
			
// IDENTIFIERS

	PHP_FUNCTION		=	[\_a-zA-Z][\w_]*
			(css:function)
	PHP_IDENTIFIER		=	\$[\w_]+
			(css:identifier)
	PHP_SPECIAL_VARS	=	(\$(GLOBALS|_(COOKIE|ENV|FILES|GET|(PO|REQUE)ST|SE(RVER|SSION))))
			(css:special)
	PHP_KEYWORD			=	break|case|continue|declare|default|do|each|elseif|else|foreach|for|goto|if|include|include_once|require|require_once|return|switch|while
			(css:keyword)
	PHP_LITERAL			=	true|TRUE|false|FALSE|null|NULL
			(css:literal)
	PHP_RESERVED		=	exception|as|var|class|const|die|echo|empty|eval|exit|extends|function|global|isset|list|new|print|static|unset|use|__FUNCTION__|__CLASS__|__METHOD__|final|interface|implements|extends|public|private|protected|abstract|clone|\$this
			(css:keyword)

	PHP_ARITHMETIC_OP		=	\+\+?|\-\-?|\*|\%|\/
			(css:operator)
	PHP_ASSIGNMENT_OP		=	\+=|\-=|\*=|\/=|\.=|\%=|\&=|\|=|\^=|<<=|>>=|=>?
			(css:operator)
	PHP_BITWISE_OP		=	&|\||\^|\~|<<|>>
			(css:operator)

			(css:operator)
	PHP_ERROR_CONTROL_OP	=	@
			(css:operator)
	PHP_LOGICAL_OP		=	and|or|xor|!|&&|\|\|
			(css:operator)
	PHP_STRING_OP			=	\.
			(css:operator)
	PHP_TYPE_OP			=	\(int\)|\(float\)|\(string\)|\(array\)|\(object\)|\(bool\)
			(css:operator)

	ELISION		=	\,
